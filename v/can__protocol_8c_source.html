171char can_write_data(void)
172{
173 char status = COMMAND_SECURITY;
174
175 if((address + nb_byte_to_write) > APP_END_ADDR) // Make sure the application doesn't overwrite EEPROM or parameter space
176 return (status);
177
178 HAL_NVIC_DisableIRQ(CAN1_RX0_IRQn); // Disable CAN interrupt until write completes
179
180 // New Implementation (Doesn't require cpt_var)
181 // Number of bytes to write should never exceecd the size of the buffer
182 if(nb_byte_to_write)
183 {
184 memcpy(&buffer[pt_buffer], RxData, RxHeader.DLC);
185 pt_buffer += RxHeader.DLC;
186 nb_byte_to_write -= RxHeader.DLC;
187 }
188
189 if (nb_byte_to_write)
190 {
191 status = COMMAND_NEW_DATA;
192 }
193 else if (nb_byte_to_write == 0 && pt_buffer != 0)
194 {
195 HAL_FLASH_Unlock();
196
197 for (int i = 0; i < pt_buffer; i += 8) // Program into flash in double word increments (8 bytes)
198 {
199 if (HAL_FLASH_Program(TYPEPROGRAM_DOUBLEWORD, address + i, *(uint64_t *)&buffer[i]) != HAL_OK)
200 {
201 status = COMMAND_SECURITY;
202 }
203 else
204 {
205 status = COMMAND_OK;
206 }
207 }
208
209 HAL_FLASH_Lock(); // Re-lock the flash afterwards
210
211 memset(buffer, 0xFF, sizeof buffer); // Clear the buffer
212 }
213
214 HAL_NVIC_EnableIRQ(CAN1_RX0_IRQn); // Enable CAN interrupt
215 return (status);
216}
